💾 Persistent Data Storage

def load_json(fn, default):
    """Load JSON with auto-creation if missing"""
    if not os.path.exists(fn):
        with open(fn, "w") as f: 
            json.dump(default, f, indent=2)
        return default
    try:
        with open(fn, "r") as f: 
            return json.load(f)
    except: 
        return default

def check_hierarchy(author, target):
    """Prevent moderating users above your role"""
    if target.id == author.guild.owner_id: 
        return False
    if author.id == author.guild.owner_id: 
        return True
    return author.top_role > target.top_role
