        await log_chan.send(embed=log_embed, file=transcript)

def load_json(fn, default):
    if not os.path.exists(fn):
        with open(fn, "w") as f: 
            json.dump(default, f, indent=2)
        return default
    
    try:
        with open(fn, "r") as f: 
            return json.load(f)
    except: 
        return default

def save_ticket_rating(channel_name, rating):
    ratings = load_json(RATINGS_FILE, {})
    ratings[channel_name] = rating
    save_json(RATINGS_FILE, ratings)

def check_hierarchy(author, target):
    if target.id == author.guild.owner_id: 
        return False
    if author.id == author.guild.owner_id: 
        return True
    return author.top_role > target.top_role

async def log_moderation(guild, action, member, moderator, reason=None):

class AnnouncementModal(Modal, title="📝 Announcement Builder"):
    title_input = TextInput(
        label="Title", 
        required=True
    )
    description_input = TextInput(
        label="Description", 
        style=discord.TextStyle.long
    )
    color_input = TextInput(
        label="Color", 
        required=False
    )
    
    async def on_submit(self, interaction):
        embed = discord.Embed(
            title=self.title_input.value,
            description=self.description_input.value,
            color=COLOR_SECONDARY
        )
        ANNOUNCEMENT_CACHE[interaction.user.id]["embed"] = embed
